const parseCSV = text => {
  const rows=[]; let row=[], field='', inQ=false;
  for(let i=0;i<text.length;i++){
    const c=text[i], n=text[i+1];
    if(c==='"'){if(inQ&&n==='"'){field+='"';i++;}else inQ=!inQ;}
    else if(c===','&&!inQ){row.push(field.trim());field='';}
    else if((c==='\n'||(c==='\r'&&n==='\n'))&&!inQ){
      if(c==='\r')i++;
      row.push(field.trim());
      if(row.some(f=>f)) rows.push(row);
      row=[];field='';
    } else field+=c;
  }
  if(field||row.length){row.push(field.trim());if(row.some(f=>f))rows.push(row);}
  return rows;
};

const parseTrainingCSV = rows => {
  // rows[2] = header row: corner | c01\nName | c02\nName | ...
  // rows[3]+ = data: "p01 — Villa Foca" | x | | x | ...
  if(rows.length<3) return {};
  const headers=rows[2]||[];
  // Extract cleaner IDs from header (format "c01\nAnita A." or "c01")
  const clIds=headers.slice(1).map(h=>{
    const m=(h||'').match(/^(c\d+)/);
    return m?m[1]:null;
  }).filter(Boolean);

  const training={};
  rows.slice(3).forEach(row=>{
    if(!row[0]) return;
    // Extract property ID from "p01 — Villa Foca"
    const pm=row[0].match(/^(p\d+)/);

// ── CONFLICT ENGINE ─────────────────────────────────────────────────────────────
const detectConflicts = jobs => {
  const out = new Map();
  const add = (id, issue) => { if(!out.has(id)) out.set(id,[]); out.get(id).push(issue); };
  jobs.forEach(job => {
    const prop=PROPERTIES.find(p=>p.id===job.propertyId); if(!prop) return;
    const jS=t2m(job.startTime), jE=jS+prop.duration*60;
    if(prop.requires2&&job.cleanerIds.length<2)
      add(job.id,{type:'requires2',msg:`Richiede 2 cleaners — assegnata solo ${job.cleanerIds.length}`});
    job.cleanerIds.forEach(cId=>{
      const cl=CLEANERS.find(c=>c.id===cId); if(!cl) return;
      if(!cl.days.includes(job.day)) add(job.id,{type:'unavailable',cId,msg:`${cl.name}: non disponibile ${DAY_IT[job.day]}`});
      if(!TRAINING[cId]?.includes(job.propertyId)) add(job.id,{type:'untrained',cId,msg:`${cl.name}: non formata per ${prop.name}`});
      // FIX T04: report overlap on earlier job only; tie-break on id for same start time
      jobs.filter(o=>o.id!==job.id&&o.day===job.day&&o.cleanerIds.includes(cId)&&
        (t2m(o.startTime)>jS||(t2m(o.startTime)===jS&&o.id>job.id))
      ).forEach(o=>{
        const oP=PROPERTIES.find(p=>p.id===o.propertyId); if(!oP) return;
        const oS=t2m(o.startTime), oE=oS+oP.duration*60;
        if(jS<oE&&jE>oS) add(job.id,{type:'overlap',cId,msg:`${cl.name}: già a ${oP.name} (${o.startTime}–${m2t(oE)})`});
      });
    });
  });
  return out;
};

    'mercoledì':'Wed','mercoledi':'Wed','mer':'Wed',
    'giovedì':'Thu','giovedi':'Thu','gio':'Thu',
    'venerdì':'Fri','venerdi':'Fri','ven':'Fri',
    'sabato':'Sat','sab':'Sat',
    'domenica':'Sun','dom':'Sun',
  };

  // Context words that signal unavailability
  const OFF_CTX = ['riposa','riposo','fuori','non disponibile','assente','off','pausa','libera','libero'];

  // For each day name found, check if preceded (within 40 chars) by an off-context word
  Object.entries(DAY_MAP).forEach(([it,en])=>{
    const idx = t.indexOf(it);
    if(idx === -1) return;
    const before = t.slice(Math.max(0,idx-40), idx);
    if(OFF_CTX.some(ctx => before.includes(ctx)) || t.includes(`riposa ${it}`) || t.includes(`fuori ${it}`)){
      if(!result.daysOff.includes(en)){
        result.daysOff.push(en);
        result.parsed.push(`Non disponibile ${DAY_FULL[en]}`);
      }
    }
  });

  // Max daily hours: "max 6 ore al giorno", "max 6 ore giornaliere", "max 6h"
  const dailyM = t.match(/max(?:imo)?\s+(\d+(?:[.,]\d+)?)\s*(?:h\b|ore?\s*(?:al\s*giorno|giornali)?)/);
  if(dailyM){ result.maxDailyHrs=parseFloat(dailyM[1].replace(',','.')); result.parsed.push(`Max ${result.maxDailyHrs}h/giorno`); }

  // Max weekly hours: "max 30 ore questa settimana" / "max 30 ore settimanali"
  const weeklyM = t.match(/max(?:imo)?\s+(\d+)\s*ore?\s*(?:questa\s*)?settimana(?:li)?/);
  if(weeklyM){ result.maxWeeklyHrs=parseInt(weeklyM[1]); result.parsed.push(`Max ${result.maxWeeklyHrs}h/settimana`); }

  // Max end time: "max fino alle 15", "fino alle 15:30", "solo mattina" → 13:00
  const endM = t.match(/(?:max\s+)?fino\s+alle\s+(\d{1,2})(?::(\d{2}))?/);
  if(endM){ result.maxEndTime=`${endM[1].padStart(2,'0')}:${endM[2]||'00'}`; result.parsed.push(`Fine max ${result.maxEndTime}`); }
  else if(t.includes('solo mattina')||t.includes('solo la mattina')){ result.maxEndTime='13:00'; result.parsed.push('Solo mattina (max 13:00)'); }

  return result;
};

// Check if a "Valida per settimana" date range covers the current week
const isWeekValid = (weekStr, weekOffset) => {
  if(!weekStr?.trim()) return false;
  const dates = getWeekDates(weekOffset);
  const weekMon = dates[0]; const weekSun = dates[6];
  // Parse dates like "19/05/2026" or "19/05"
  const parts = weekStr.split(/[–\-—]/);
  const parseDate = s => {
    const clean = s.trim();
    const m = clean.match(/(\d{1,2})[\/\-\.](\d{1,2})(?:[\/\-\.](\d{4}))?/);
    if(!m) return null;
    const year = m[3] ? parseInt(m[3]) : weekMon.getFullYear();
    return new Date(year, parseInt(m[2])-1, parseInt(m[1]));
  };
  if(parts.length < 2) return false;
  const from = parseDate(parts[0]); const to = parseDate(parts[1]);
  if(!from||!to) return false;
  // Overlap: week range overlaps with validity range
  return from <= weekSun && to >= weekMon;
};

// Build constraint map: { cId: parsedResult } — only for valid weeks
const buildConstraints = (weekOffset) => {
  const out = {};
  CLEANERS.forEach(cl => {
    if(!cl.extraRequests?.trim()) return;
    const valid = !cl.extraWeek?.trim() || isWeekValid(cl.extraWeek, weekOffset);
    if(!valid) return;
    const parsed = parseExtraRequests(cl.extraRequests, weekOffset);
    if(parsed.daysOff.length||parsed.maxDailyHrs||parsed.maxWeeklyHrs||parsed.maxEndTime)
      out[cl.id] = { ...parsed, cleanerName: cl.name };
  });
  return out;
};

// ── AUTO-SCHEDULER — soft limits + richieste extra ─────────────────────────────
const runAutoSchedule = (allJobs, weekOffset=0) => {
  const constraints = buildConstraints(weekOffset);

  const weeklyHrs = Object.fromEntries(CLEANERS.map(cl=>[cl.id,0]));
  const dailyHrs  = Object.fromEntries(CLEANERS.map(cl=>[cl.id,Object.fromEntries(DAYS.map(d=>[d,0]))]));
  const daySched  = Object.fromEntries(CLEANERS.map(cl=>[cl.id,Object.fromEntries(DAYS.map(d=>[d,[]]))]));

  // Commit already-assigned jobs
  allJobs.filter(j=>j.cleanerIds.length>0).forEach(job=>{
    const prop=PROPERTIES.find(p=>p.id===job.propertyId); if(!prop) return;
    job.cleanerIds.forEach(cId=>{
      weeklyHrs[cId]=(weeklyHrs[cId]||0)+prop.duration;
      dailyHrs[cId]?.[job.day] && (dailyHrs[cId][job.day]+=prop.duration);
      daySched[cId]?.[job.day]?.push({start:t2m(job.startTime),end:t2m(job.startTime)+prop.duration*60});
    });
  });

  const unassigned=[...allJobs.filter(j=>j.cleanerIds.length===0)].sort((a,b)=>{
    const dd=DAYS.indexOf(a.day)-DAYS.indexOf(b.day);
    return dd!==0?dd:t2m(a.startTime)-t2m(b.startTime);
  });

  const newlyScheduled = unassigned.map(job=>{
    const prop=PROPERTIES.find(p=>p.id===job.propertyId);
    if(!prop) return {...job,status:'needs_review'};
    const needed=prop.cleanersRequired||( prop.requires2?2:1);
    const jS=t2m(job.startTime), jE=jS+prop.duration*60;

    const eligible=CLEANERS
      .filter(cl=>TRAINING[cl.id]?.includes(job.propertyId))
      .filter(cl=>cl.days.includes(job.day))
      // HARD: skip if daysOff in constraints
      .filter(cl=>!(constraints[cl.id]?.daysOff||[]).includes(job.day))
      // HARD: skip if time overlap
      .filter(cl=>!(daySched[cl.id]?.[job.day]||[]).some(s=>jS<s.end&&jE>s.start))
      // HARD: skip if maxEndTime would be exceeded
      .filter(cl=>{
        const maxEnd=constraints[cl.id]?.maxEndTime;
        return !maxEnd || jE <= t2m(maxEnd);
      })
      // SOFT: sort — prefer under daily limit, then under weekly limit, then fewest hours
      .sort((a,b)=>{
        const aOver = (constraints[a.id]?.maxDailyHrs && (dailyHrs[a.id]?.[job.day]||0)+prop.duration > constraints[a.id].maxDailyHrs) ? 1 : 0;
        const bOver = (constraints[b.id]?.maxDailyHrs && (dailyHrs[b.id]?.[job.day]||0)+prop.duration > constraints[b.id].maxDailyHrs) ? 1 : 0;
        if(aOver!==bOver) return aOver-bOver; // prefer under daily limit
        const aWOver = (constraints[a.id]?.maxWeeklyHrs && (weeklyHrs[a.id]||0)+prop.duration > constraints[a.id].maxWeeklyHrs) ? 1 : 0;
        const bWOver = (constraints[b.id]?.maxWeeklyHrs && (weeklyHrs[b.id]||0)+prop.duration > constraints[b.id].maxWeeklyHrs) ? 1 : 0;
        if(aWOver!==bWOver) return aWOver-bWOver; // prefer under weekly limit
        return (weeklyHrs[a.id]||0)-(weeklyHrs[b.id]||0); // fewest hours
      });

    const pick=eligible.slice(0,needed);
    pick.forEach(cl=>{
      weeklyHrs[cl.id]=(weeklyHrs[cl.id]||0)+prop.duration;
      dailyHrs[cl.id]?.[job.day] && (dailyHrs[cl.id][job.day]+=prop.duration);
      daySched[cl.id]?.[job.day]?.push({start:jS,end:jE});
    });

    // Mark soft-limit warnings
    const warnings=pick.filter(cl=>{
      const c=constraints[cl.id];
      return c && (
        (c.maxDailyHrs && (dailyHrs[cl.id]?.[job.day]||0) > c.maxDailyHrs) ||
        (c.maxWeeklyHrs && (weeklyHrs[cl.id]||0) > c.maxWeeklyHrs)
      );
    }).map(cl=>cl.name);

    return {
      ...job,
      cleanerIds:pick.map(c=>c.id),
      status:pick.length>=needed?'assigned':'needs_review',
      softLimitWarning: warnings.length>0 ? `Supera limite: ${warnings.join(', ')}` : null,
    };
  });

  return {
    jobs: [...allJobs.filter(j=>j.cleanerIds.length>0).map(j=>({...j,status:'assigned'})), ...newlyScheduled],
    constraints,
  };
};

// ── PARSE AI ACTIONS ──────────────────────────────────────────────────────────
// AI embeds [AZIONI]...[/AZIONI] blocks in responses when suggesting changes.
const parseAIActions = (text) => {
  const match = text.match(/\[AZIONI\]([\s\S]*?)\[\/AZIONI\]/);
  if (!match) return { displayText: text, suggestions: [] };
  const displayText = text.replace(/\[AZIONI\][\s\S]*?\[\/AZIONI\]/, '').trim();
  try {
    const parsed = JSON.parse(match[1].trim());
    return { displayText, suggestions: parsed.suggestions || [] };
  } catch { return { displayText: text, suggestions: [] }; }
};

  const doLoadTemplate=(merge=false)=>{
    const newJobs=SHEET_TEMPLATE.map(r=>({...r,id:uid(),status:'assigned'}));
    setJobs(prev=>{
      let base=merge?prev:[];
      // FIX: deduplication — skip rows already matching propertyId+day+startTime
      const toAdd=newJobs.filter(nj=>!base.some(e=>e.propertyId===nj.propertyId&&e.day===nj.day&&e.startTime===nj.startTime));
      return [...base,...toAdd];
    });
    // Safeguard: if shifts already generated, mark stale
    setWf(w=>({...w,templateLoaded:true,staleGeneration:w.shiftsGenerated?true:w.staleGeneration}));
    setModal(null);
    showToast(`✓ Template caricato${merge?' (merge)':''}`);
  };

  // ── HOSTAWAY IMPORT (with stale guard) ──
  const doImportHostaway=(rows)=>{
    setJobs(prev=>{
      const toAdd=rows.map(r=>({...r,id:uid(),status:'unassigned'})).filter(nj=>!prev.some(e=>e.propertyId===nj.propertyId&&e.day===nj.day&&e.startTime===nj.startTime));
      return [...prev,...toAdd];
    });
    // Safeguard: if shifts already generated, mark stale
    setWf(w=>({...w,hostawayLoaded:true,staleGeneration:w.shiftsGenerated?true:w.staleGeneration}));
    setModal(null);
    showToast(`✓ ${rows.length} checkout importati da Hostaway`);
  };

  // ── AUTO-SCHEDULE ──
  const handleGenerate=(startDay, endDay)=>{
    if(jobs.length===0){showToast('Carica prima il template o importa da Hostaway','warn');return;}
    if(!startDay){ setGenerateResult(null); setModal('generateResult'); return; }
    try{
      const startIdx=DAYS.indexOf(startDay);
      const endIdx  =endDay?DAYS.indexOf(endDay):6;
      const keepBefore=jobs.filter(j=>DAYS.indexOf(j.day)<startIdx);
      const keepAfter =jobs.filter(j=>DAYS.indexOf(j.day)>endIdx);
      const regenJobs =jobs.filter(j=>{const i=DAYS.indexOf(j.day);return i>=startIdx&&i<=endIdx;});
      const stripped=regenJobs.map(j=>({...j,cleanerIds:[],status:'unassigned'}));
      const result=runAutoSchedule([...keepBefore,...stripped,...keepAfter], weekOffset);
      const finalJobs=[
        ...keepBefore.map(j=>({...j,status:'assigned'})),
        ...result.jobs.filter(j=>{const i=DAYS.indexOf(j.day);return i>=startIdx&&i<=endIdx;}),
        ...keepAfter.map(j=>({...j,status:'assigned'})),
      ];
      setGenerateResult({...result,jobs:finalJobs,startDay,endDay:endDay||'Sun'});
      setModal('generateResult');
    }catch(e){
      console.error('runAutoSchedule error:',e);
      showToast('Errore nella generazione: '+e.message,'error');
    }
  };

  const applyGenerate=()=>{
    if(!generateResult) return;
    setJobs(generateResult.jobs);
    setWf(w=>({...w,shiftsGenerated:true,staleGeneration:false}));
    setModal(null);
    setGenerateResult(null);
    const nr=generateResult.jobs.filter(j=>j.status==='needs_review').length;
    const sw=generateResult.jobs.filter(j=>j.softLimitWarning).length;
    const msg = nr>0
      ? '⚡ Turni generati · '+nr+' da rivedere'+(sw>0?' · '+sw+' limite soft':'')
      : sw>0 ? '⚡ Turni generati · '+sw+' con limite soft' : '⚡ Tutti i turni assegnati!';
    showToast(msg,(nr>0||sw>0)?'warn':'ok');
  };

  // Apply AI-suggested actions directly to jobs
  const applySuggestion=useCallback((actions)=>{
    if(!actions?.length) return;
    setJobs(prev=>{
      let updated=[...prev];
      actions.forEach(act=>{
        if(act.type==='add'){
          // Don't add duplicates
          const exists=updated.some(j=>j.propertyId===act.propertyId&&j.day===act.day&&j.startTime===act.startTime);
          if(!exists) updated=[...updated,{id:uid(),propertyId:act.propertyId,day:act.day,startTime:act.startTime,cleanerIds:act.cleanerIds||[],note:act.note||'AI suggerimento',type:'extra',status:'assigned'}];
        } else if(act.type==='reassign'){
          updated=updated.map(j=>j.propertyId===act.propertyId&&j.day===act.day&&j.startTime===act.startTime?{...j,cleanerIds:act.cleanerIds||[]}:j);
        } else if(act.type==='remove'){
          updated=updated.filter(j=>!(j.propertyId===act.propertyId&&j.day===act.day&&j.startTime===act.startTime));
        }
      });
      return updated;
    });
    setWf(w=>({...w,staleGeneration:w.shiftsGenerated}));
    showToast(`✓ ${actions.length} ${actions.length===1?'modifica applicata':'modifiche applicate'} dall'AI`);
  },[showToast]);

  const goToConflictDay=useCallback((day)=>{
    setSelDay(day);setTab('day');setHighlightJob(null);
    if(conflictNavTimer.current)clearTimeout(conflictNavTimer.current);
    conflictNavTimer.current=setTimeout(()=>{const f=jobs.find(j=>j.day===day&&conflicts.has(j.id));if(f)setHighlightJob(f.id);},50);
  },[jobs,conflicts]);

  const sendChat=async()=>{
    if(!chatIn.trim()||aiLoad) return;
    const apiKey = GEMINI_KEY !== 'INSERISCI_QUI_LA_TUA_KEY' ? GEMINI_KEY : localStorage.getItem('cf_gemini_key');
    if(!apiKey){ setModal('apikey'); return; }
    const txt=chatIn.trim();setChatIn('');
    setMsgs(p=>[...p,{role:'user',text:txt}]);setAiLoad(true);

    const callGemini=async(retries=2,delayMs=1500)=>{
      const sysPrompt=buildSystemPrompt(jobs,weekOffset);
      const recentHistory=msgs.slice(-10);
      const contents=recentHistory.map(m=>({
        role:m.role==='assistant'?'model':'user',
        parts:[{text:m.text.replace(/\[AZIONI\][\s\S]*?\[\/AZIONI\]/g,'').trim()}]
      }));
      contents.push({role:'user',parts:[{text:txt}]});
      const res=await fetch(
        'https://generativelanguage.googleapis.com/v1beta/models/gemini-3.5-flash:generateContent?key='+apiKey,
        {method:'POST',headers:{'Content-Type':'application/json'},
          body:JSON.stringify({
            system_instruction:{parts:[{text:sysPrompt}]},
            contents,
            generationConfig:{maxOutputTokens:1200},
          })}
      );
      if((res.status===429||res.status===403)&&retries>0){
        await new Promise(r=>setTimeout(r,delayMs));
        return callGemini(retries-1,delayMs*1.5);
      }
      if(!res.ok) throw new Error(res.status===400?'API key non valida — controlla nelle impostazioni.':('Errore '+res.status));
      const data=await res.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text||'Risposta vuota.';
    };

    try{
      const reply=await callGemini();
      setMsgs(p=>[...p,{role:'assistant',text:reply}]);
    }catch(e){
      setMsgs(p=>[...p,{role:'assistant',text:e.message||'Errore di rete. Riprova.'}]);
    }
    setAiLoad(false);
  };

  const fixConflict=(job,errs)=>{const prop=PROPERTIES.find(p=>p.id===job.propertyId);setChatIn(`Ho un conflitto ${DAY_FULL[job.day]} per ${prop?.name} (${job.startTime}):\n${errs.map(e=>'• '+e).join('\n')}\n\nCome lo risolvo?`);setTab('chat');};
  const openAdd=day=>{setEditJob(null);setSelDay(day);setModal('add');};
  const openEdit=job=>{setEditJob(job);setModal('edit');};
  const saveJob=form=>{
    if(editJob){setJobs(p=>p.map(j=>j.id===editJob.id?{...form,id:editJob.id}:j));showToast('✓ Turno aggiornato');}
    else{setJobs(p=>[...p,{...form,id:uid()}]);showToast('✓ Turno aggiunto');}
    setModal(null);
  };
  const deleteJob=()=>{
    if(!editJob) return;
    const snapshot={...editJob,cleanerIds:[...editJob.cleanerIds]};
    const prop=PROPERTIES.find(p=>p.id===snapshot.propertyId);
    setJobs(p=>p.filter(j=>j.id!==snapshot.id));
    setModal(null);
    showToast(`🗑 ${prop?.name||'Turno'} ${DAY_IT[snapshot.day]} eliminato`,'warn',()=>{setJobs(p=>[...p,snapshot]);setToast(null);});
  };

  const NAV=[{id:'week',icon:'📅',label:'Sett.'},{id:'day',icon:'📋',label:DAY_IT[selDay]},{id:'props',icon:'🏢',label:'Prop.'},{id:'chat',icon:'🤖',label:'AI'},{id:'cleaners',icon:'👥',label:'Team'}];

  // ── LOGIN SCREEN — after all hooks ────────────────────────────────────────
  if(!authed) return(
    <div style={{minHeight:'100vh',background:C.nav,display:'flex',alignItems:'center',justifyContent:'center',fontFamily:"'Nunito',sans-serif"}}>
      <style>{`@import url('https://fonts.googleapis.com/css2?family=Nunito:wght@400;600;700;800&family=Syne:wght@700;800&display=swap');*{box-sizing:border-box;margin:0;padding:0}`}</style>
      <div style={{background:'white',borderRadius:24,padding:'40px 36px',width:'100%',maxWidth:360,boxShadow:'0 20px 60px rgba(0,0,0,0.3)',textAlign:'center'}}>
        <div style={{fontFamily:"'Syne',sans-serif",fontSize:28,fontWeight:800,color:C.nav,marginBottom:4}}>CleanFlow</div>
        <div style={{fontSize:13,color:C.muted,marginBottom:32}}>Scheduling · Malta</div>
        <input type="password" value={pwInput}
          onChange={e=>{setPwInput(e.target.value);setPwError(false);}}
          onKeyDown={e=>{if(e.key==='Enter'){if(pwInput==='172390'){sessionStorage.setItem('cf_auth','1');setAuthed(true);}else setPwError(true);}}}
          placeholder="Password" autoFocus
          style={{width:'100%',padding:'13px 16px',borderRadius:12,border:`1.5px solid ${pwError?C.conflict:'#E0DDD8'}`,fontSize:16,textAlign:'center',letterSpacing:4,fontFamily:'inherit',marginBottom:10,outline:'none'}}
        />
        {pwError&&<div style={{fontSize:12,color:C.conflict,fontWeight:700,marginBottom:10}}>Password errata</div>}
        <button onClick={()=>{if(pwInput==='172390'){sessionStorage.setItem('cf_auth','1');setAuthed(true);}else setPwError(true);}}
          style={{width:'100%',padding:'13px',borderRadius:12,background:C.nav,color:'white',fontWeight:800,fontSize:15,border:'none',cursor:'pointer',fontFamily:'inherit'}}>
          Accedi
        </button>
      </div>
    </div>
  );

  const mainContent=(
    <div style={{padding:'16px 16px 0',flex:1,overflowY:'auto',minHeight:0}}>
      {tab==='week'&&<WeekView jobs={jobs} conflicts={conflicts} weekOffset={weekOffset} onDaySelect={d=>{setSelDay(d);setTab('day');}} onConflictDay={goToConflictDay} onLoadTemplate={()=>setModal('template')} onImportHostaway={()=>setModal('hostaway')} onGenerate={handleGenerate} wf={wf}/>}
      {tab==='day'&&<DayView jobs={jobs} conflicts={conflicts} day={selDay} setDay={d=>{setSelDay(d);setHighlightJob(null);}} weekOffset={weekOffset} onAdd={openAdd} onEdit={openEdit} onFix={fixConflict} highlightJobId={highlightJob} setHighlightJob={setHighlightJob} onLoadTemplate={()=>setModal('template')}/>}
      {tab==='props'&&<PropertiesView jobs={jobs} conflicts={conflicts} weekOffset={weekOffset} onDayAndJobSelect={(day,jobId)=>{setSelDay(day);setHighlightJob(jobId);setTab('day');}}/>}
      {tab==='chat'&&<ChatView msgs={msgs} chatIn={chatIn} setChatIn={setChatIn} onSend={sendChat} isLoading={aiLoad} chatRef={chatRef} onApplySuggestion={applySuggestion}/>}
      {tab==='cleaners'&&<CleanersView jobs={jobs} selDay={selDay} weekOffset={weekOffset} onDaySelect={d=>setSelDay(d)}/>}
    </div>
  );

  return(
    <div style={{background:C.bg,minHeight:'100vh',fontFamily:"'Nunito',sans-serif",display:'flex',flexDirection:'column'}}>
      <style>{`@import url('https://fonts.googleapis.com/css2?family=Nunito:wght@400;600;700;800&family=Syne:wght@700;800&display=swap');*{box-sizing:border-box;margin:0;padding:0}button{font-family:'Nunito',sans-serif;outline:none}textarea,input,select{font-family:'Nunito',sans-serif;outline:none}::-webkit-scrollbar{width:4px;height:4px}::-webkit-scrollbar-thumb{background:#ccc;border-radius:2px}`}</style>
      <Toast toast={toast}/>

      {isDesktop?(
        // ── DESKTOP LAYOUT ──
        <div style={{display:'flex',height:'100vh',overflow:'hidden'}}>
          <Sidebar tab={tab} setTab={setTab} selDay={selDay} weekOffset={weekOffset} setWeekOffset={setWeekOffset} wf={wf} totalC={totalC} jobs={jobs} conflicts={conflicts} onGenerate={handleGenerate} onConflictDay={goToConflictDay} setModal={setModal}/>
          <div style={{flex:1,display:'flex',flexDirection:'column',overflow:'hidden'}}>
            {/* Sheet status bar */}
            {sheetStatus==='loading'&&<div style={{background:'#F0F7FF',borderBottom:'1px solid #BFDBFE',padding:'6px 24px',fontSize:11,color:'#2A5FA0',display:'flex',alignItems:'center',gap:6}}>
              <span style={{animation:'spin 1s linear infinite',display:'inline-block'}}>⟳</span> Caricamento dati da Google Sheet...
              <style>{`@keyframes spin{from{transform:rotate(0deg)}to{transform:rotate(360deg)}}`}</style>
            </div>}
            {sheetStatus==='ok'&&sheetInfo&&<div style={{background:'#F0FFF4',borderBottom:'1px solid #86EFAC',padding:'5px 24px',fontSize:11,color:'#166534',display:'flex',alignItems:'center',gap:6}}>
              ✓ Sheet: {sheetInfo}
            </div>}
            {sheetStatus==='error'&&<div style={{background:'#FFFBEB',borderBottom:'1px solid #FCD34D',padding:'5px 24px',fontSize:11,color:C.amber,display:'flex',alignItems:'center',gap:6}}>
              ⚠️ {sheetInfo||'Sheet non raggiungibile — dati demo attivi'}
            </div>}
            {/* Stale warning desktop */}
            {wf.staleGeneration&&<div style={{background:'#FFF3CD',borderBottom:'1px solid #FFCC02',padding:'10px 24px',display:'flex',alignItems:'center',gap:10}}>
              <span>⚠️</span>
              <span style={{fontSize:13,fontWeight:700,color:'#856404',flex:1}}>Dati aggiornati dopo la generazione — rigenera i turni prima di esportare</span>
              <button onClick={()=>handleGenerate()} style={{background:'#856404',color:'white',border:'none',borderRadius:8,padding:'6px 14px',fontSize:12,fontWeight:800,cursor:'pointer',fontFamily:'inherit'}}>⚡ Rigenera Turni</button>
            </div>}
            <div style={{flex:1,overflowY:'auto',maxWidth:720,margin:'0 auto',width:'100%',padding:'24px 24px 80px'}}>
              {mainContent}
            </div>
          </div>
        </div>
      ):(
        // ── MOBILE LAYOUT ──
        <>
          <div style={{background:C.nav,padding:'14px 16px 12px',position:'sticky',top:0,zIndex:100}}>
            <div style={{display:'flex',alignItems:'center',justifyContent:'space-between',marginBottom:10}}>
              <div style={{fontFamily:"'Syne',sans-serif",fontSize:22,fontWeight:800,color:'white',letterSpacing:'-0.5px'}}>CleanFlow</div>
              <div style={{display:'flex',gap:8,alignItems:'center'}}>
                {totalC>0&&<button onClick={()=>goToConflictDay(jobs.find(j=>conflicts.has(j.id))?.day||'Mon')} style={{background:C.conflict,color:'white',borderRadius:20,padding:'4px 10px',fontSize:12,fontWeight:800,border:'none',cursor:'pointer'}}>⚠️ {totalC}</button>}
                <button onClick={()=>{setApiKeyInput(localStorage.getItem('cf_gemini_key')||'');setModal('apikey');}} style={{background:'rgba(255,255,255,0.15)',color:'white',borderRadius:10,padding:'7px 10px',fontSize:15,border:'none',cursor:'pointer'}}>{localStorage.getItem('cf_gemini_key')?'⚙️':'🔑'}</button>
                <button onClick={()=>setModal('wiw')} style={{background:'#217346',color:'white',borderRadius:10,padding:'7px 13px',fontSize:12,fontWeight:800,border:'none',cursor:'pointer',display:'flex',alignItems:'center',gap:5}}>
                  <span>📊</span><span>WIW</span>
                </button>
              </div>
            </div>
            <div style={{display:'flex',alignItems:'center',justifyContent:'space-between',background:'rgba(255,255,255,0.08)',borderRadius:10,padding:'7px 12px'}}>
              <button onClick={()=>setWeekOffset(w=>w-1)} style={{background:'transparent',border:'none',color:'white',fontSize:20,cursor:'pointer',padding:'0 6px',opacity:0.8}}>‹</button>
              <div style={{textAlign:'center'}}>
                <div style={{fontSize:13,color:'white',fontWeight:700}}>{fmtWeek(weekOffset)}</div>
                {weekOffset===0?<div style={{fontSize:10,color:C.accent,fontWeight:700}}>Settimana corrente</div>:<button onClick={()=>setWeekOffset(0)} style={{fontSize:10,color:'#7A9AB5',background:'transparent',border:'none',cursor:'pointer',fontFamily:'inherit'}}>← Torna a questa settimana</button>}
              </div>
              <button onClick={()=>setWeekOffset(w=>w+1)} style={{background:'transparent',border:'none',color:'white',fontSize:20,cursor:'pointer',padding:'0 6px',opacity:0.8}}>›</button>
            </div>
          </div>

          <WorkflowBanner wf={wf} onLoadTemplate={()=>setModal('template')} onImportHostaway={()=>setModal('hostaway')} onGenerate={handleGenerate} totalC={totalC} isDesktop={false}/>

          <div style={{flex:1,overflowY:'auto',paddingBottom:80}}>
            {mainContent}
          </div>

          <div style={{position:'fixed',bottom:0,left:0,right:0,background:C.nav,display:'flex',borderTop:'1px solid #2E3F50',zIndex:100}}>
            {NAV.map(t=>(
              <button key={t.id} onClick={()=>setTab(t.id)} style={{flex:1,padding:'9px 2px 12px',background:'transparent',color:tab===t.id?C.accent:'#5A7A90',borderTop:tab===t.id?`2px solid ${C.accent}`:'2px solid transparent',border:'none',cursor:'pointer',display:'flex',flexDirection:'column',alignItems:'center',gap:2}}>
                <span style={{fontSize:17}}>{t.icon}</span>
                <span style={{fontSize:9,fontWeight:tab===t.id?800:500}}>{t.label}</span>
              </button>
            ))}
          </div>
        </>
      )}

      {/* MODALS */}
      {modal==='apikey'&&<Modal onClose={()=>setModal(null)}>
        <div style={{fontFamily:"'Syne',sans-serif",fontSize:19,fontWeight:800,color:C.text,marginBottom:4}}>🔑 Impostazioni AI</div>
        <div style={{fontSize:12,color:C.muted,marginBottom:16}}>Inserisci la tua Gemini API key per attivare l'assistente AI.</div>
        <div style={{background:'#F0F7FF',borderRadius:10,padding:'12px 14px',marginBottom:16,fontSize:12,color:'#2A5FA0',lineHeight:1.6}}>
          <div style={{fontWeight:800,marginBottom:4}}>Come ottenerla gratis:</div>
          1. Vai su <strong>ai.google.dev</strong><br/>
          2. Clicca "Get API key" → Google AI Studio<br/>
          3. Create API key → copia la chiave
        </div>
        <div style={{fontSize:11,fontWeight:800,color:C.muted,letterSpacing:1,textTransform:'uppercase',marginBottom:8}}>Gemini API Key</div>
        <input
          type="password"
          value={apiKeyInput}
          onChange={e=>setApiKeyInput(e.target.value)}
          placeholder="AIza..."
          style={{width:'100%',padding:'11px 14px',borderRadius:10,border:'1.5px solid #E0DDD8',fontSize:14,fontFamily:'monospace',marginBottom:8}}
        />
        <div style={{fontSize:11,color:C.muted,marginBottom:20}}>La chiave viene salvata solo nel tuo browser (localStorage). Non viene mai inviata ai nostri server.</div>
        <div style={{display:'flex',gap:8}}>
          {localStorage.getItem('cf_gemini_key')&&<button onClick={()=>{localStorage.removeItem('cf_gemini_key');setApiKeyInput('');setModal(null);showToast('API key rimossa','warn');}} style={{padding:'11px 14px',borderRadius:10,background:'#FDE8E8',color:C.conflict,fontWeight:700,fontSize:13,border:'none',cursor:'pointer',fontFamily:'inherit'}}>Rimuovi</button>}
          <button onClick={()=>setModal(null)} style={{flex:1,padding:'12px',borderRadius:10,background:'#F0EDE8',color:C.text,fontWeight:700,fontSize:13,border:'none',cursor:'pointer',fontFamily:'inherit'}}>Annulla</button>
          <button onClick={()=>{if(apiKeyInput.trim()){localStorage.setItem('cf_gemini_key',apiKeyInput.trim());setModal(null);showToast('✓ API key salvata — AI attiva');}}} disabled={!apiKeyInput.trim()} style={{flex:2,padding:'12px',borderRadius:10,background:apiKeyInput.trim()?C.nav:'#DDD',color:'white',fontWeight:800,fontSize:13,border:'none',cursor:apiKeyInput.trim()?'pointer':'default',fontFamily:'inherit'}}>Salva e Attiva</button>
        </div>
      </Modal>}

      {modal==='template'&&<Modal onClose={()=>setModal(null)}>
        {/* Inline template modal */}
        <div style={{fontFamily:"'Syne',sans-serif",fontSize:19,fontWeight:800,color:C.text,marginBottom:4}}>📊 Carica Default Template</div>
        <div style={{display:'flex',alignItems:'center',gap:8,background:'#F0FFF4',border:'1px solid #86EFAC',borderRadius:10,padding:'10px 14px',marginBottom:14}}>
          <span>📊</span><div><div style={{fontSize:12,fontWeight:800,color:'#166534'}}>Google Sheets · Foglio "Template"</div><div style={{fontSize:11,color:'#16a34a'}}>Demo — dati simulati</div></div>
        </div>
        {wf.shiftsGenerated&&<div style={{background:'#FFF3CD',border:'1px solid #FFCC02',borderRadius:10,padding:'12px 14px',marginBottom:14}}>
          <div style={{fontSize:13,fontWeight:800,color:'#856404',marginBottom:4}}>⚠️ Turni già generati</div>
          <div style={{fontSize:12,color:'#92400E'}}>Caricare il template segnerà i turni come da rigenerare.</div>
        </div>}
        {jobs.length>0&&!wf.shiftsGenerated&&<div style={{background:'#FFF3CD',border:'1px solid #FFCC02',borderRadius:10,padding:'12px 14px',marginBottom:14}}>
          <div style={{fontSize:13,fontWeight:800,color:'#856404',marginBottom:4}}>Hai già turni questa settimana</div>
          <div style={{fontSize:12,color:'#92400E'}}>Sostituisci tutto o aggiungi (merge) i turni template.</div>
        </div>}
        <div style={{background:'#F9F8F5',borderRadius:12,padding:'4px 0',marginBottom:20,maxHeight:200,overflowY:'auto'}}>
          {DAYS.filter(day=>SHEET_TEMPLATE.some(r=>r.day===day)).map(day=>(
            <div key={day}>
              <div style={{fontSize:10,fontWeight:800,color:C.muted,letterSpacing:1,textTransform:'uppercase',padding:'8px 14px 4px'}}>{DAY_FULL[day]}</div>
              {SHEET_TEMPLATE.filter(r=>r.day===day).map((r,i)=>{const p=PROPERTIES.find(x=>x.id===r.propertyId);const cls=r.cleanerIds.map(id=>CLEANERS.find(c=>c.id===id)?.name.split(' ')[0]).filter(Boolean);return(<div key={i} style={{display:'flex',alignItems:'center',gap:10,padding:'7px 14px',borderBottom:'1px solid #F0EDE8'}}><span>{TI(p?.type)}</span><div style={{flex:1}}><div style={{fontSize:13,fontWeight:700,color:C.text}}>{p?.name}</div><div style={{fontSize:11,color:C.muted}}>{r.startTime} · {cls.join(' + ')}{r.note?' · '+r.note:''}</div></div></div>);})}
            </div>
          ))}
        </div>
        <div style={{display:'flex',gap:8}}>
          <button onClick={()=>setModal(null)} style={{flex:1,padding:'12px',borderRadius:10,background:'#F0EDE8',color:C.text,fontWeight:700,fontSize:13,border:'none',cursor:'pointer',fontFamily:'inherit'}}>Annulla</button>
          {jobs.length>0&&<button onClick={()=>doLoadTemplate(true)} style={{flex:1,padding:'12px',borderRadius:10,background:'white',color:C.nav,fontWeight:700,fontSize:13,border:`2px solid ${C.nav}`,cursor:'pointer',fontFamily:'inherit'}}>+ Merge</button>}
          <button onClick={()=>doLoadTemplate(false)} style={{flex:jobs.length>0?1:2,padding:'12px',borderRadius:10,background:C.nav,color:'white',fontWeight:800,fontSize:13,border:'none',cursor:'pointer',fontFamily:'inherit'}}>{jobs.length>0?'Sostituisci':'✓ Carica'}</button>
        </div>
      </Modal>}

      {modal==='hostaway'&&<Modal onClose={()=>setModal(null)}><HostawayModal onImport={doImportHostaway} onClose={()=>setModal(null)}/></Modal>}
      {modal==='generateResult'&&<Modal onClose={()=>{setModal(null);setGenerateResult(null);}}><GenerateModal result={generateResult} constraints={generateResult?.constraints} onConfirm={applyGenerate} onClose={()=>{setModal(null);setGenerateResult(null);}} onGenerate={(startDay)=>handleGenerate(startDay)}/></Modal>}
      {modal==='add'&&<Modal onClose={()=>setModal(null)}><JobForm initial={{propertyId:'',day:selDay,startTime:'09:00',cleanerIds:[],note:'',type:'extra'}} onSave={saveJob}/></Modal>}
      {modal==='edit'&&editJob&&<Modal onClose={()=>setModal(null)}><JobForm initial={{propertyId:editJob.propertyId,day:editJob.day,startTime:editJob.startTime,cleanerIds:[...editJob.cleanerIds],note:editJob.note||'',type:editJob.type||'extra'}} onSave={saveJob} onDelete={deleteJob}/></Modal>}
      {modal==='wiw'&&<Modal onClose={()=>setModal(null)}><WIWExportModal jobs={jobs} weekOffset={weekOffset} wf={wf}/></Modal>}
    </div>
  );
}

    const root = ReactDOM.createRoot(document.getElementById('root'));
    root.render(<App />);
  </script>
